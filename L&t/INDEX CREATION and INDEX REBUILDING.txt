INDEX CREATION and INDEX REBUILDING

Index creation:- 


Create index index_name on table_name (coloumn_1,coloumn_2,….…);

select OWNER,INDEX_NAME,TABLE_NAME,BLEVEL,STATUS from dba_indexes where table_name=’&tab_name’ and index_name=’&index_name’;

Index rebuilding;

alter index schema.index_name rebuild online parallel (integer);

Or

alter index schema.index_name rebuild online noparallel (integer);

e.g :- 

create index.sql

ALTER INDEX  LTF_LEA_MIL.SYS_C0064899 REBUILD online parallel 10;
ALTER INDEX  LTF_LEA_MIL.SYS_C0064899 REBUILD noparallel;
ALTER INDEX  LTF_LEA_MIL.LRBH_USERID_UPLOADDATE_IDX REBUILD online parallel 10;
ALTER INDEX  LTF_LEA_MIL.LRBH_USERID_UPLOADDATE_IDX REBUILD noparallel;
